Class Cookie

A module to handle cookies through a store.

Usage:

import { Cookie } from '@queelag/core'

interface AuthData {
  token: string
}

class AuthStore {
  data: AuthData

  constructor() {
    this.data = { token: '' }
    Cookie.get(this.name, this.data)
  }

  login(): void {
    let response: string

    response = 'token'
    this.token = response

    Cookie.set(this.name, this.data, ['token'], { secure: true })
  }

  logout(): void {
    Cookie.remove(this.name, this.data)
  }

  get name(): string {
    return 'auth'
  }
}
